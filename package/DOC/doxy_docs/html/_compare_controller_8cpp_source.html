244 wxRect CompareController::getCropRectangle()
245 {
246  const float zoom = sharedData->imageInfo.zoom;
247  const float coeff = 100.f / zoom;
248 
249  int x1 = sharedData->cropData.x;
250  int x2 = sharedData->cropData.x0;
251  int y1 = sharedData->cropData.y;
252  int y2 = sharedData->cropData.y0;
253 
254  return {
255  wxPoint{static_cast<int>(((x1 > x2 ? x2 : x1) - sharedData->imageInfo.x) * coeff),
256  static_cast<int>(((y1 > y2 ? y2 : y1) - sharedData->imageInfo.y) * coeff)},
257  wxPoint{static_cast<int>(((x1 > x2 ? x1 : x2) - sharedData->imageInfo.x) * coeff),
258  static_cast<int>(((y1 > y2 ? y1 : y2) - sharedData->imageInfo.y) * coeff)}
259  };
